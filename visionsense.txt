// Label smoothing — only accept a label confirmed in 2 of last 3 frames
const LABEL_HISTORY_SIZE = 3;
let labelHistory = [];

function getMostStableLabel(detections) {
    if (!detections.length) { labelHistory = []; return null; }
    const top = detections[0];
    labelHistory.push(top.label);
    if (labelHistory.length > LABEL_HISTORY_SIZE) labelHistory.shift();
    // Count frequency
    const freq = {};
    for (const l of labelHistory) freq[l] = (freq[l] || 0) + 1;
    const best = Object.entries(freq).sort((a,b) => b[1]-a[1])[0];
    // Require label to appear in at least 2 of last 3 frames
    return best[1] >= 2 ? best[0] : null;
}

function setStatus(message) { statusEl.textContent = message; }

function showError(err) {
    const message = err?.message || String(err);
    errorBox.style.display = "block";
    errorBox.innerHTML = `<strong>ERROR:</strong> ${escapeHtml(message)}<br><br>
        <strong>What to try:</strong><br>
        1. Run this page through a local server (e.g. VS Code Live Server).<br>
        2. Keep internet connected while the model downloads.<br>
        3. Switch to Fast fallback if Open Vocabulary is too large.`;
}

function escapeHtml(value) {
    return value.replace(/[&<>"']/g, c =>
        ({ "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#039;" }[c]));
}

function parseLabels() {
    return labelInput.value
        .split(",")
        .map(l => l.trim().toLowerCase())
        .filter(Boolean)
        .filter((l, i, arr) => arr.indexOf(l) === i)
        .slice(0, FAST_SCAN_MAX_LABELS);
}

async function setupCamera() {
    if (!navigator.mediaDevices?.getUserMedia)
        throw new Error("Camera access not available in this browser.");
    if (video.srcObject) return;
    const stream = await navigator.mediaDevices.getUserMedia({
        video: { width: { ideal: 640 }, height: { ideal: 360 }, facingMode: "environment" },
        audio: false
    });
    video.srcObject = stream;
    await new Promise((resolve, reject) => {
        video.onloadedmetadata = resolve;
        video.onerror = () => reject(new Error("Could not start webcam preview."));
    });
    await video.play();
}

async function init() {
    if (isRunning || startBtn.disabled) return;
    errorBox.style.display = "none";
    startBtn.disabled = true;
    startBtn.textContent = "STARTING CAMERA...";
    predictionEl.textContent = "Loading";
    confFill.style.width = "0%";
    try {
        activeConfig    = MODEL_CONFIGS[modelSelect.value];
        candidateLabels = parseLabels();
        if (activeConfig.usesCandidateLabels && candidateLabels.length < 2)
            throw new Error("Add at least two labels for Open Vocabulary detection.");
        await setupCamera();
        resizeCanvasToVideo();
        startBtn.textContent = "DOWNLOADING MODEL...";
        setStatus(`Loading ${activeConfig.label}. First download may take a moment.`);
        detector = await pipeline(activeConfig.task, activeConfig.id, {
            dtype: "q4", progress_callback: updateProgress
        });
        isRunning = true;
        startBtn.style.display = "none";
        setControlsLocked(true);
        setStatus(`${activeConfig.label} online. Watching for ${activeConfig.usesCandidateLabels ? candidateLabels.length + " priority labels" : "common objects"}.`);
        speak("Detection engine activated.");
        scheduleDetect(100);
    } catch (err) {
        showError(err);
        setStatus("Initialization failed. Adjust engine or labels, then retry.");
        predictionEl.textContent = "System Idle";
        startBtn.disabled = false;
        startBtn.textContent = "RETRY INITIALIZATION";
        setControlsLocked(false);
    }
}

function updateProgress(progress) {
    if (progress.status === "progress" && progress.total) {
        const pct = Math.max(0, Math.min(100, Math.round((progress.loaded / progress.total) * 100)));
        startBtn.textContent = `DOWNLOADING: ${pct}%`;
    } else if (progress.status === "ready") {
        startBtn.textContent = "FINALIZING ENGINE...";
    }
}

function setControlsLocked(locked) {
    modelSelect.disabled = locked;
    labelInput.disabled  = locked;

function resizeCanvasToVideo() {
    const w = video.videoWidth || 1280;
    const h = video.videoHeight || 720;
    if (canvas.width !== w || canvas.height !== h) { canvas.width = w; canvas.height = h; }
}

function scheduleDetect(delay = activeConfig?.interval || 800) {
    clearTimeout(detectTimer);
    detectTimer = setTimeout(detect, delay);
}

async function detect() {
    if (!isRunning || !detector || isDetecting) return;
    isDetecting = true;
    try {
        resizeCanvasToVideo();
        const frame  = await captureFrame();
        const output = activeConfig.usesCandidateLabels
            ? await detector(frame.image, candidateLabels, { threshold: activeConfig.threshold, top_k: 6 })
            : await detector(frame.image, { threshold: activeConfig.threshold, top_k: 6 });
        skippedFrames = 0;
        errorBox.style.display = "none";
        const detections = normalizeDetections(output, frame.scaleX, frame.scaleY);
        drawDetections(detections);
        updatePrediction(detections);
    } catch (err) {
        console.warn("Detection frame skipped:", err);
        skippedFrames++;
        setStatus(`Skipped ${skippedFrames} frame${skippedFrames === 1 ? "" : "s"} while catching up.`);
        if (skippedFrames >= 3) {
            predictionEl.textContent = "Frame Error";
            confFill.style.width = "0%";
            showError(err);
            isRunning = false;
            startBtn.style.display = "block";
            startBtn.disabled = false;
            startBtn.textContent = activeConfig.usesCandidateLabels ? "TRY FAST FALLBACK" : "RETRY INITIALIZATION";
            if (activeConfig.usesCandidateLabels) modelSelect.value = "fast-coco";
            setControlsLocked(false);
            labelInput.disabled = modelSelect.value !== "open-vocab";
        }
    } finally {
        isDetecting = false;
        scheduleDetect();
    }
}

async function captureFrame() {
    if (video.readyState < 2) throw new Error("Webcam frame not ready yet.");
    const sourceWidth  = video.videoWidth  || 640;
    const sourceHeight = video.videoHeight || 360;
    const maxWidth = activeConfig.usesCandidateLabels ? 320 : 416;
    const scale  = Math.min(1, maxWidth / sourceWidth);
    const width  = Math.max(1, Math.round(sourceWidth  * scale));
    const height = Math.max(1, Math.round(sourceHeight * scale));
    if (frameCanvas.width !== width || frameCanvas.height !== height) {
        frameCanvas.width = width; frameCanvas.height = height;
    }
    frameCtx.drawImage(video, 0, 0, width, height);
    const rawImage = RawImage.fromCanvas
        ? RawImage.fromCanvas(frameCanvas)
        : await RawImage.fromURL(frameCanvas.toDataURL("image/jpeg", 0.86));
    return {
        image: rawImage.rgb ? rawImage.rgb() : rawImage,
        scaleX: canvas.width / width,
        scaleY: canvas.height / height
    };
}

function normalizeDetections(output, scaleX = 1, scaleY = 1) {
    if (!Array.isArray(output)) return [];
    return output
        .filter(item => item?.box && Number.isFinite(item.score))
        .map(item => ({
            label: item.label || "object",
            score: item.score,
            box: {
                xmin: clamp(item.box.xmin * scaleX, 0, canvas.width),
                ymin: clamp(item.box.ymin * scaleY, 0, canvas.height),
                xmax: clamp(item.box.xmax * scaleX, 0, canvas.width),
                ymax: clamp(item.box.ymax * scaleY, 0, canvas.height)
            }
        }))
        .filter(item => item.box.xmax > item.box.xmin && item.box.ymax > item.box.ymin)
        .sort((a, b) => b.score - a.score)
        .slice(0, 6);
}

function clamp(value, min, max) {
    return Math.max(min, Math.min(max, Number(value) || 0));
}

function drawDetections(detections) {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.font = "bold 15px Arial";
    ctx.textBaseline = "top";
    detections.forEach(obj => {
        // Mirror the box horizontally to match the CSS-flipped webcam
        const mirrorXmin = canvas.width - obj.box.xmax;
        const mirrorXmax = canvas.width - obj.box.xmin;
        const { ymin, ymax } = obj.box;
        const w = mirrorXmax - mirrorXmin;
        const h = ymax - ymin;
        const label = `${obj.label} ${Math.round(obj.score * 100)}%`;
        const textW  = Math.min(ctx.measureText(label).width + 12, canvas.width - mirrorXmin);
        const labelY = ymin > 28 ? ymin - 28 : ymin + 4;

        ctx.strokeStyle = "#facc15"; ctx.lineWidth = 3;
        ctx.strokeRect(mirrorXmin, ymin, w, h);
        ctx.fillStyle = "#facc15";
        ctx.fillRect(mirrorXmin, labelY, textW, 24);
        ctx.fillStyle = "#000";
        // Text is drawn in normal (un-flipped) canvas — reads correctly
        ctx.fillText(label, mirrorXmin + 6, labelY + 4, Math.max(20, textW - 10));
    });
}

function updatePrediction(detections) {
    if (!detections.length) {
        predictionEl.textContent = "Scanning";
        confFill.style.width = "0%";
        stableHits = 0;
        lastSpoken = "";
        lastDisposalLabel = "";
        labelHistory = [];
        return;
    }

    const top = detections[0];
    const stableLabel = getMostStableLabel(detections);
    const displayLabel = stableLabel || top.label;

    predictionEl.textContent = displayLabel;
    confFill.style.width = `${Math.round(top.score * 100)}%`;
    setStatus(`${detections.length} object${detections.length === 1 ? "" : "s"} found. Confidence: ${Math.round(top.score * 100)}%`);

    // stableHits tracks how many consecutive frames the SAME label has been seen
    if (displayLabel === lastSpoken) {
        stableHits++;
    } else {
        stableHits = 1;
        lastDisposalLabel = ""; // different object — allow disposal card to refresh
    }
    lastSpoken = displayLabel;

    const now = Date.now();
    if (stableHits === 1 && top.score >= activeConfig.threshold && now - lastSpeechAt > 2200) {
        speak(displayLabel);
        lastSpeechAt = now;
    }

    // Show disposal card once label is stable for 2+ frames AND not already shown
    if (stableHits >= 2 && displayLabel !== lastDisposalLabel) {
        fetchDisposalGuide(displayLabel);
    }
}

const DEFAULT_DISPOSAL = {
    category: "trash",
    steps: [
        "Check if your local council has a specific collection for this item",
        "If unsure, place in general waste to avoid contaminating recycling",
        "Search Earth911.com or your local waste authority website for guidance",
        "Consider if the item could be repaired, donated, or repurposed first"
    ],
    tip: "When in doubt, reduce and reuse before worrying about recycling."
};

function lookupDisposal(rawLabel) {
    const label = rawLabel.toLowerCase().trim();
    // Direct match
    if (DISPOSAL_DB[label]) return DISPOSAL_DB[label];
    // Keyword map match
    const mapped = KEYWORD_MAP[label];
    if (mapped && DISPOSAL_DB[mapped]) return DISPOSAL_DB[mapped];
    // Partial match — check if any DB key appears inside the label or vice versa
    for (const key of Object.keys(DISPOSAL_DB)) {
        if (label.includes(key) || key.includes(label)) return DISPOSAL_DB[key];
    }
    for (const [kw, mapped2] of Object.entries(KEYWORD_MAP)) {
        if (label.includes(kw) || kw.includes(label)) return DISPOSAL_DB[mapped2] || null;
    }
    return null;
}

function fetchDisposalGuide(label) {
    if (label === lastDisposalLabel) return;
    lastDisposalLabel = label;

    disposalCard.style.display    = "block";
    disposalLoading.style.display = "none";
    disposalBody.style.display    = "block";

    const data = lookupDisposal(label) || DEFAULT_DISPOSAL;
    renderDisposalCard(label, data);
}

function renderDisposalCard(label, data) {
    const cat   = (data.category || "trash").toLowerCase();
    const style = CATEGORY_STYLES[cat] || CATEGORY_STYLES.trash;
    const steps = Array.isArray(data.steps) ? data.steps : [];

    const borderColors = {
        recycle:"rgba(74,222,128,0.35)",
        trash:"rgba(248,113,113,0.35)",
        hazard:"rgba(251,146,60,0.35)",
        compost:"rgba(96,165,250,0.35)",
        donate:"rgba(250,204,21,0.35)"
    };
    disposalCard.style.borderColor = borderColors[cat] || "#333";

    disposalTitle.textContent = `How to dispose: ${label}`;
    disposalIcon.textContent  = style.icon;

    disposalBadge.className   = `disposal-badge ${style.badge}`;
    disposalBadge.textContent = style.label;

    disposalSteps.innerHTML = steps.map((s, i) => `
        <div class="step">
            <span class="step-num">${String(i + 1).padStart(2,"0")}.</span>
            <span class="step-text">${escapeHtml(String(s))}</span>
        </div>
    `).join("");

    if (data.tip) {
        disposalSteps.innerHTML += `
            <div style="margin-top:12px;padding:10px 12px;background:#111;border-radius:6px;border:1px solid #1e1e1e;">
                <span style="color:#555;font-size:0.7rem;letter-spacing:1px;">ECO TIP · </span>
                <span style="color:#888;font-size:0.78rem;">${escapeHtml(String(data.tip))}</span>
            </div>`;
    }

    disposalLoading.style.display = "none";
    disposalBody.style.display    = "block";

    speak(`${label}. ${style.label}. ${steps[0] || ""}`);
}



function speak(text) {
    if (!("speechSynthesis" in window)) return;
    window.speechSynthesis.cancel();
    window.speechSynthesis.speak(new SpeechSynthesisUtterance(text));
}
